import Vue from 'vue'
import Vuex from 'vuex'

Vue.use(Vuex)

export default new Vuex.Store({
  state: {
    artists: [],
    artist: {
      id: null,
      nombre: null
    }
  },
  mutations: {
    addArtist(state){
      state.artists = [state.artist,...state.artists]
    }
  },
  actions: {
    actionAddArtist(context){
      context.commit('addArtist')
    }
  },
  getters: {
    getAllArtist:(state)=>state.artists
  }
})
